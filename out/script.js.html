let i = 0;
let recipes = [];


function addToRecipes(name) {
    let title = document.getElementById('title');
    let description = document.getElementById('description');
    let recipe = document.getElementById('recipe');


    let recipy = {
        "title": title.value,
        "description": description.value
    }

    recipes.push(recipy);


    for (i; i < recipe.children.length; i++) { }

    recipe.innerHTML += `
    <div id="subDiv${i}" style="display: flex; align-items: center;">
        <div style="padding: 8px;">
            <div>${recipy['title']}</div>
            <div>${recipy['description']}</div>
        </div>
        <div onclick="deleteItems(${i})" style="padding: 8px;"><a href="#">del</a></div>
    </div>`;

    title.value = "";
    description.value = "";
}

function deleteItems(i) {

}
